void STACK::push(char c)
{
	top++;
	a[top] = c;
	a[top+1]='\0';
    
	int i;

}
void STACK::palindrome(){
    char str1[15];
    int i,j;
    for(i=top,j=0; i>=0; i--,j++)
	{
		str1[j]=a[i];
	}
	str1[j]='\0';

    if(strcmp(str1,a) == 0)
		cout<<"\n\nString is palindrome";
	else
		cout<<"\n\nString is not palindrome";

}
