
#include <iostream>
using namespace std;

struct queue
{       
	int Que[15];
	int front,rear;
};

class Queue
{    
	struct queue q;
    public:
      Queue(){
		q.front=q.rear=-1;
		}
      void enqueue(int);
      int delqueue();
      void display();
};

void Queue::enqueue(int x)
{
	q.Que[++q.rear]=x;
}
int Queue::delqueue()
{
	return q.Que[++q.front];
}

void Queue::display()
{   int i;
    cout<<"\n";
    for(i=q.front+1;i<=q.rear;i++)
	     cout<<q.Que[i]<<" ";
}
